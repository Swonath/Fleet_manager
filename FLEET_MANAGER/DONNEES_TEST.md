?? REQUÊTES UTILES POUR TESTER:

-- Voir tous les véhicules avec stats
SELECT v.*, 
       COUNT(c.id_carburant) as nb_ravitaillements,
       SUM(c.quantite_litres) as total_litres
FROM vehicules v
LEFT JOIN carburants c ON v.id_vehicule = c.id_vehicule
GROUP BY v.id_vehicule;

-- Consommation moyenne par véhicule
SELECT v.marque, v.modele,
       SUM(c.quantite_litres) as total_litres,
       SUM(t.distance_parcourue) as total_km,
       ROUND(SUM(t.distance_parcourue) / SUM(c.quantite_litres), 2) as km_par_litre
FROM vehicules v
LEFT JOIN carburants c ON v.id_vehicule = c.id_vehicule
LEFT JOIN trajets t ON v.id_vehicule = t.id_vehicule
GROUP BY v.id_vehicule;

-- Coûts totaux par mois
SELECT DATE_FORMAT(c.date_saisie, '%Y-%m') as mois,
       SUM(c.cout_total) as cout_total,
       COUNT(DISTINCT c.id_vehicule) as vehicules
FROM carburants c
GROUP BY DATE_FORMAT(c.date_saisie, '%Y-%m');

-- Maintenances à venir
SELECT m.*, v.marque, v.modele
FROM maintenances m
JOIN vehicules v ON m.id_vehicule = v.id_vehicule
WHERE m.date_prochaine_maintenance IS NOT NULL
AND m.date_prochaine_maintenance > NOW()
ORDER BY m.date_prochaine_maintenance;
